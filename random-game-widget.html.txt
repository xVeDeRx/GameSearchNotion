// ── Proxy (bypasses Notion CORS) ──────────────────────────────────────────
const PROXY = "https://corsproxy.io/?url=";

function notionUrl(path) {
  return PROXY + encodeURIComponent("https://api.notion.com/v1" + path);
}

// ── Storage helpers ───────────────────────────────────────────────────────
function save(key, val) { try { localStorage.setItem(key, JSON.stringify(val)); } catch(e){} }
function load(key) { try { return JSON.parse(localStorage.getItem(key)); } catch(e){ return null; } }

// ── Config ────────────────────────────────────────────────────────────────
function getConfig() { return load("ngw_config") || {}; }

function saveConfig(c) { save("ngw_config", c); }

// ── Cache helpers (daily/weekly) ─────────────────────────────────────────
function getCachedGame(mode) {
  if (mode === "always") return null;
  const cached = load("ngw_cache");
  if (!cached) return null;
  const now = Date.now();
  const age = now - cached.ts;
  if (mode === "daily" && age < 86400000) return cached.game;
  if (mode === "weekly" && age < 604800000) return cached.game;
  return null;
}

function setCachedGame(game) { save("ngw_cache", { ts: Date.now(), game }); }

// ── API call ──────────────────────────────────────────────────────────────
async function fetchAllGames(apiKey, dbId) {
  let games = [], cursor = undefined, tries = 0;
  while (true) {
    const body = { page_size: 100 };
    if (cursor) body.start_cursor = cursor;
    const res = await fetch(notionUrl(`/databases/${dbId}/query`), {
      method: "POST",
      headers: {
        "Authorization": "Bearer " + apiKey,
        "Notion-Version": "2022-06-28",
        "Content-Type": "application/json"
      },
      body: JSON.stringify(body)
    });
    if (!res.ok) {
      const err = await res.json().catch(() => ({}));
      throw new Error(err.message || "API error " + res.status);
    }
    const data = await res.json();
    games = games.concat(data.results);
    if (!data.has_more || ++tries > 10) break;
    cursor = data.next_cursor;
  }
  return games;
}

// ── Pick a random entry ───────────────────────────────────────────────────
function pickRandom(arr) { return arr[Math.floor(Math.random() * arr.length)]; }

function extractText(prop) {
  if (!prop) return "";
  if (prop.type === "title") return prop.title.map(t => t.plain_text).join("") || "";
  if (prop.type === "rich_text") return prop.rich_text.map(t => t.plain_text).join("") || "";
  if (prop.type === "select") return prop.select?.name || "";
  if (prop.type === "multi_select") return prop.multi_select.map(s => s.name).join(", ");
  if (prop.type === "number") return prop.number?.toString() || "";
  if (prop.type === "date") return prop.date?.start || "";
  if (prop.type === "checkbox") return prop.checkbox ? "✓" : "";
  if (prop.type === "url") return prop.url || "";
  return "";
}

function extractCover(page) {
  if (!page.cover) return null;
  if (page.cover.type === "external") return page.cover.external.url;
  if (page.cover.type === "file") return page.cover.file.url;
  return null;
}

// ── Render widget ─────────────────────────────────────────────────────────
function renderGame(game, config) {
  const nameProp = config.nameProp || "Name";
  const chipProp = config.chipProp;

  const name = extractText(game.properties[nameProp]) || "Unknown";
  const cover = extractCover(game);
  const url = game.url;

  document.getElementById("w-name").textContent = name;
  document.getElementById("w-name").classList.add("fade");

  // Cover
  const coverEl = document.getElementById("w-cover");
  const ph = document.getElementById("w-placeholder");
  coverEl.innerHTML = "";
  if (cover) {
    const img = document.createElement("img");
    img.src = cover;
    img.alt = name;
    img.onerror = () => { coverEl.innerHTML = `<div class="cover-placeholder">${name[0]}</div>`; };
    coverEl.appendChild(img);
  } else {
    ph.textContent = name[0] || "?";
    coverEl.appendChild(ph);
  }
  coverEl.classList.add("fade");

  // Meta chips
  const meta = document.getElementById("w-meta");
  meta.innerHTML = "";
  if (chipProp && game.properties[chipProp]) {
    const val = extractText(game.properties[chipProp]);
    if (val) {
      val.split(", ").forEach(v => {
        const c = document.createElement("span");
        c.className = "chip accent";
        c.textContent = v;
        meta.appendChild(c);
      });
    }
  }

  // Link
  document.getElementById("w-link").href = url || "#";

  // Tag label
  const mode = config.refreshMode;
  document.getElementById("w-tag").textContent =
    mode === "daily" ? "🎮 Game of the Day" :
    mode === "weekly" ? "🎮 Game of the Week" :
    "🎮 Random Game";
}

// ── Main load logic ───────────────────────────────────────────────────────
async function loadWidget(config, forceReroll) {
  document.getElementById("setup").style.display = "none";
  document.getElementById("widget").style.display = "block";

  if (!forceReroll) {
    const cached = getCachedGame(config.refreshMode);
    if (cached) { renderGame(cached, config); return; }
  }

  // Show loading state
  document.getElementById("w-name").textContent = "Loading...";
  document.getElementById("w-meta").innerHTML = "";
  document.getElementById("w-cover").innerHTML = `<div class="cover-placeholder shimmer" style="width:100%;height:100%"></div>`;

  try {
    const games = await fetchAllGames(config.apiKey, config.dbId);
    if (!games.length) throw new Error("No entries found in database.");
    const game = pickRandom(games);
    setCachedGame(game);
    renderGame(game, config);
  } catch (err) {
    document.getElementById("w-name").textContent = "Error";
    document.getElementById("w-meta").innerHTML = `<span style="color:#ff6b6b;font-size:13px">${err.message}</span>`;
  }
}

async function reroll() {
  const config = getConfig();
  await loadWidget(config, true);
}

function showSetup() {
  const c = getConfig();
  document.getElementById("api-key").value = c.apiKey || "";
  document.getElementById("db-id").value = c.dbId || "";
  document.getElementById("name-prop").value = c.nameProp || "Name";
  document.getElementById("chip-prop").value = c.chipProp || "";
  document.getElementById("refresh-mode").value = c.refreshMode || "always";
  document.getElementById("widget").style.display = "none";
  document.getElementById("setup").style.display = "block";
}

async function saveAndLoad() {
  const apiKey = document.getElementById("api-key").value.trim();
  const dbId = document.getElementById("db-id").value.trim();
  const nameProp = document.getElementById("name-prop").value.trim() || "Name";
  const chipProp = document.getElementById("chip-prop").value.trim();
  const refreshMode = document.getElementById("refresh-mode").value;
  const errEl = document.getElementById("error-msg");
  errEl.textContent = "";

  if (!apiKey || !dbId) { errEl.textContent = "API key and Database ID are required."; return; }

  const config = { apiKey, dbId, nameProp, chipProp, refreshMode };
  saveConfig(config);
  await loadWidget(config, true);
}

// ── Auto-start if config saved ────────────────────────────────────────────
window.addEventListener("DOMContentLoaded", () => {
  const config = getConfig();
  if (config.apiKey && config.dbId) {
    loadWidget(config, false);
  }
});
